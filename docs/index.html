// Your first Zap program
let name = "World"
print(`Hello, {name}!`)

fn greet(person) {
  return `Welcome to Zap, {person}!`
}

print(greet("Mukesh"))

// Loop 1 to 5
loop i from 1 to 6 {
  print(`Count: {i}`)
}
// Full-stack HTTP server
import time from "zap:time"

server on 3000 {

  get "/" {
    return {
      message: "Zap API ⚡",
      uptime: time.dateLocal()
    }
  }

  get "/hello/:name" {
    let n = req.params.name
    return { greeting: `Hello, {n}!` }
  }

  post "/api/echo" {
    return { ok: true, data: req.body }
  }

}
// Compiles to a real native binary
fn fibonacci(n) {
  if (n <= 1) { return n }
  return fibonacci(n-1) + fibonacci(n-2)
}

let scores = [85, 92, 78, 96, 88]
print(scores.sort())
print(scores.join(", "))

loop i from 0 to 10 {
  print(fibonacci(i))
}

Why Zap?

Everything you need.
Nothing you don't.

Blazing Fast

Compiles to C99 for native speed, or runs on Node.js for instant startup. You choose per project.

Clean Syntax

No semicolons, no boilerplate. Reads like English — easier than JavaScript, cleaner than Python.

Full-Stack Ready

Built-in server block handles HTTP routing. Build REST APIs with zero framework setup.

Batteries Included

8 stdlib modules out of the box: fs, http, os, math, time, crypto, str, arr. No npm needed.

Secure by Default

Built-in crypto module: UUID, SHA-256/512, HMAC, safe comparison, tokens, base64.

VS Code Extension

Syntax highlighting, snippets, hover docs, auto-complete and file icons for .zap / .zp files.

Quick Start

Up and running in 30 seconds.

01

Install

npm install -g zap-language
02

Create a file

echo 'print("Hello!")' > app.zap
03

Run it

zap run app.zap
04

Build native binary

zap build app.zap --native
