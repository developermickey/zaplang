Installation

Zap requires Node.js 16+ for the JS runtime. For native compilation, clang or gcc must be installed.

npm install -g zap-language

Verify installation:

zap --version

CLI Commands

zap run

Transpile and run a .zap file using Node.js.

zap run hello.zap

zap build

Compile to JavaScript.

zap build hello.zap

zap build --native

Compile to a native binary via C99 + clang/gcc.

zap build hello.zap --native
zap build hello.zap --native -o myapp

zap emit-c

Emit the generated C code (for inspection/debugging).

zap emit-c hello.zap

zap fmt

Auto-format a .zap file in-place. Use --check to verify formatting without writing (useful in CI).

zap fmt hello.zap            # format file in-place
zap fmt .                    # format all .zap/.zp files recursively
zap fmt hello.zap --check    # exit 1 if not formatted

zap repl

Start the interactive REPL. Context persists across inputs — define variables and functions, then use them.

zap repl

REPL commands:

  • .help — show available commands
  • .reset — reset the session context
  • .clear — clear the terminal
  • .exit — quit the REPL

zap new

Scaffold a new Zap project.

zap new myproject

Variables

Declare variables with let. Zap is dynamically typed — no type annotations needed.

let name = "Zap"
let version = 1.1
let active = true
let nothing = null

// Reassign
version = 2.0

// Compound assignment
version += 0.1
version -= 0.1
version *= 2
version /= 2

Functions

fn add(a, b) {
  return a + b
}

print(add(2, 3))  // 5

// Anonymous function
let double = fn(x) { return x * 2 }
print(double(5))   // 10

// Ternary expression
let max = a > b ? a : b

Control Flow

let score = 85

if (score >= 90) {
  print("A")
} else if (score >= 80) {
  print("B")
} else {
  print("C")
}

Loops

// Range loop
loop i from 0 to 5 {
  print(i)
}

// Loop with step
loop i from 0 to 10 step 2 {
  print(i)  // 0 2 4 6 8
}

// While loop
let n = 0
while (n < 5) {
  print(n)
  n++
}

// break / continue
loop i from 0 to 10 {
  if (i == 5) { break }
  if (i % 2 == 0) { continue }
  print(i)
}

Arrays

let nums = [1, 2, 3, 4, 5]
let words = ["hello", "world"]

// Access
print(nums[0])       // 1

// Methods
nums.push(6)
let last = nums.pop()
print(nums.len())
print(nums.reverse())
print(nums.sort())
print(nums.includes(3))   // true
print(nums.indexOf(3))    // 2
print(nums.join(","))      // 1,2,3,4,5
print(nums.slice(1, 3))   // [2, 3]

// Range
let r = range(1, 6)       // [1,2,3,4,5]

Objects

Objects are key-value maps. They work in both JS and native compilation modes.

let person = {
  name: "Mukesh",
  age: 25,
  active: true
}

// Property access
print(person.name)      // Mukesh

// Property assignment
person.age = 26
print(person)           // {name: Mukesh, age: 26, active: true}

// Object methods
let keys = person.keys()    // ["name", "age", "active"]
let vals = person.values()  // ["Mukesh", 26, true]

// Dynamic key access
person.set("email", "zap@example.com")
let email = person.get("email")

Strings & Templates

let name = "Zap"

// Template string (backtick)
let msg = `Hello, {name}! Version {1 + 1}`
print(msg)   // Hello, Zap! Version 2

// Built-in string functions
print(upper("hello"))     // HELLO
print(lower("WORLD"))     // world
print(len("hello"))       // 5
print(trim("  hi  "))     // hi

Error Handling

try {
  let data = riskyOperation()
  print(data)
} catch (err) {
  print(`Error: {err}`)
} finally {
  print("Done")
}

Server Block

server on 3000 {

  get "/" {
    return "Hello from Zap!"
  }

  get "/user/:id" {
    return { id: req.params.id }
  }

  post "/api/echo" {
    return req.body
  }

  put    "/api/update" { return { ok: true } }
  delete "/api/item"   { return { deleted: true } }

}

Native Compilation

Zap can compile to a native binary via C99. Requires clang or gcc.

# Compile to binary
zap build app.zap --native

# Custom output name
zap build app.zap --native -o myapp

# Run the binary
./myapp

Native mode supports: numbers, strings, booleans, arrays, functions, loops, conditionals, and most built-ins.
