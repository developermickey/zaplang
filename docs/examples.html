Examples

Real Zap programs you can copy, run, and build on.

Hello World

let name = "World"
print(`Hello, {name}!`)

fn greet(person) {
  return `Welcome, {person}!`
}

print(greet("Mukesh"))
$ zap run hello.zap

Fibonacci

fn fibonacci(n) {
  if (n <= 1) { return n }
  return fibonacci(n - 1) + fibonacci(n - 2)
}

loop i from 0 to 10 {
  print(fibonacci(i))
}
$ zap build fibonacci.zap --native -o fib && ./fib

Working with Arrays

let scores = [85, 92, 78, 96, 88]

// Sort and reverse
print(scores.sort())
print(scores.reverse())

// Find stats
let total = 0
loop i from 0 to scores.len() {
  total += scores[i]
}
let avg = total / scores.len()
print(`Average: {avg}`)

// Filter above 90
let high = []
loop i from 0 to scores.len() {
  if (scores[i] > 90) { high.push(scores[i]) }
}
print(high.join(", "))
$ zap run arrays.zap

Web Server

import time from "zap:time"

server on 3000 {

  get "/" {
    return {
      message: "Welcome to Zap API!",
      time: time.dateLocal()
    }
  }

  get "/greet/:name" {
    let name = req.params.name
    return { greeting: `Hello, {name}!` }
  }

  post "/api/echo" {
    return { received: req.body }
  }

}
$ zap run server.zap
→ Server running at http://localhost:3000

File Operations

import fs from "zap:fs"

// Write a file
fs.writeFile("output.txt", "Hello from Zap!\n")

// Read it back
let content = fs.readFile("output.txt")
print(content)

// JSON
let data = { name: "Zap", version: 1 }
fs.writeJSON("config.json", data)
let loaded = fs.readJSON("config.json")
print(loaded.name)
$ zap run files.zap

Crypto & Security

import crypto from "zap:crypto"

// Generate UUID
print(crypto.uuid())

// Hash a password
let hash = crypto.sha256("mypassword")
print(hash)

// Random secure token
let token = crypto.randomToken(32)
print(token)

// Base64
let encoded = crypto.base64Encode("Hello Zap")
print(crypto.base64Decode(encoded))
$ zap run crypto.zap

Native — Prime Numbers

// Runs as a native binary — no Node.js needed
fn isPrime(n) {
  if (n < 2) { return false }
  loop i from 2 to n {
    if (n % i == 0) { return false }
  }
  return true
}

let count = 0
loop n from 2 to 100 {
  if (isPrime(n)) {
    print(n)
    count++
  }
}
print(`Found {count} primes`)
$ zap build primes.zp --native -o primes && ./primes
