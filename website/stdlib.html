zap:fs — File System

import fs from "zap:fs"
Function Description
fs.readFile(path)Read file as string
fs.writeFile(path, data)Write string to file
fs.readJSON(path)Read and parse JSON file
fs.writeJSON(path, obj)Stringify and write JSON
fs.readLines(path)Read file as array of lines
fs.exists(path)Check if file/dir exists
fs.mkdir(path)Create directory
fs.listDir(path)List files in directory
fs.deleteFile(path)Delete a file
fs.copyFile(src, dst)Copy a file
fs.join(...parts)Join path segments

zap:math — Mathematics

import math from "zap:math"

print(math.PI)
print(math.sqrt(144))      // 12
print(math.pow(2, 10))     // 1024
print(math.floor(3.7))     // 3
print(math.clamp(15, 0, 10)) // 10
print(math.randomInt(1, 100))
print(math.isPrime(17))    // true
print(math.factorial(5))   // 120

zap:str — String Utilities

import str from "zap:str"

print(str.upper("hello"))           // HELLO
print(str.camelCase("hello world")) // helloWorld
print(str.snakeCase("HelloWorld"))  // hello_world
print(str.isEmail("a@b.com"))       // true
print(str.words("hello world"))     // ["hello","world"]
print(str.repeat("ab", 3))         // ababab

zap:arr — Array Utilities

import arr from "zap:arr"

let nums = [3, 1, 4, 1, 5, 9]

print(arr.sum(nums))           // 23
print(arr.avg(nums))           // 3.83
print(arr.min(nums))           // 1
print(arr.max(nums))           // 9
print(arr.unique(nums))        // [3,1,4,5,9]
print(arr.shuffle(nums))
print(arr.chunk(nums, 2))     // [[3,1],[4,1],[5,9]]
print(arr.sample(nums, 3))    // 3 random items
